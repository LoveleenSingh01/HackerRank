import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {
        Scanner scan =  new Scanner(System.in);
        
        int N = 0;
        do {
           N = scan.nextInt();
        } while( N<2 && N>20);
        
        for (int i=1; i<=10; i++ )
            {
            System.out.println(+N+" x "+i+" = "+(N*i));
        }
    }
}
